 90 prototype.__init = function(args) {

 92 		if (Util.isNotNull(args['row'])) {

108 		}

110 			this._colDef = this.grid['colDefMgr'].get(Util.index(args['$'][0]));
111 		}
112 	}
113 	if (Util.isNullOr(this._datarow, this._colDef) && Util.isNotNull(args['event'])) {
114 		var node = this.grid['view']._getClosestCell(args['event'].target);
115 		if (Util.isNotNull(node)) {
