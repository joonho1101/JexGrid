 28  renderer.selectBox = function(args) {
 29 	var mappings = args['mapping'],
 30 		 attr = args['attr'],
 31 		 defaultVal = args['default'],
 32 		 style = args['style'],
 33 		 callback = args['callback'],
 34 		 label,
 35 		 defaultIdx,
 36 		 val,
 37 		 tag,
 38 		 len = 0,
 39 		 labels = [],
 40 		 values = [],
 41 		 attrStr = '<select';
 42 
 43 	// set select box attribute string
 44 	if (attr) {
 45 		for (tag in attr) {
 46 			if (attr.hasOwnProperty(tag)) {
 47 				attrStr += ' ' + tag + '="' + attr[tag] + '"';
 48 			}
 49 		}
 50 	}
 51 	if (style) {
 52 		attrStr += ' style="';
 53 		for (tag in style) {
 54 			if (style.hasOwnProperty(tag)) {
 55 				attrStr += tag + ':' + style[tag] + ';';
 56 			}
 57 		}
 58 		attrStr += '"';
 59 	}
 60 	attrStr += '>';
 61 
 62 	// find out option length, default label, and make a list of values
 63 	for (label in mappings) {
 64 		if (mappings.hasOwnProperty(label)) {
 65 			val = mappings[label];
 66 			labels.push(label);
 67 			values.push(val);
 68 			if (defaultVal == val) {
 69 				defaultIdx = len;
 70 			}
 71 			len++;
 72 		}
 73 	}
 74 
 75 	return function(value) {
 76 		var toselect,
 77 			 i,
 78 			 html = attrStr;
 79 
 80 		for (i = 0; i < len; i++) {
 81 			if (value == values[i]) {
 82 				toselect = i;
 83 				break;
 84 			}
 85 		}
 86 
 87 		if (toselect === undefined) {
 88 			toselect = defaultIdx;
 89 		}
 90 
 91 		for (i = 0; i < len; i++) {
 92 			html += '<option value="' + values[i] + '"';
 93 			if (i === toselect) {
 94 				html += ' selected="selected"';
 95 			}
 96 			html += '>' + labels[i] + '</option>';
 97 		}
 98 
 99 		html += '</select>';
100 		if (callback) {
101 			var args = [];
102 			args.push(html);
103 			Array.prototype.push.apply(args, arguments);
104 			html = callback.apply(this, args);
105 		}
106 		return html;
107 	}
108 };
