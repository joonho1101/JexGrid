 38 	if (this.length === 0) {
 39 		return false;
 40 	}

 43 		x,
 44 		y;

 46 		box = this.getBoundingRect();
 47 		x = e.pageX;

 57 			contains = true;
 58 			return false;
 59 		}
 60 	});
 61 	return contains;
