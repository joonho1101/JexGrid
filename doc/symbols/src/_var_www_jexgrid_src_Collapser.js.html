465 	prototype._onAddDatarow = function(datarow) {
466 		var node = this._tree.createNode(datarow);
467 		node._collapsed = this._options['beginCollapsed'];
468 		if (Util.isNotNull(node.parent) && (node.parent === node.tree.root || (node.parent._shown && !node.parent._collapsed))) {
469 			node._shown = true;
470 		}
471 		else {
472 			node._shown = false;
473 		}
474 
475 		if (Util.isNotNull(node.parent) && node.parent.children.length === 1) {
476 			this.grid['view'].rerenderCellByIdAndKey(this.grid['dataMgr'].getId(node.parent.data), this.key);
477 		}
478 
479 		if (node._collapsed === true || node._shown === false) {
480 			node.traverseChildren({fn:function(args) {
481 				this._shown = false;
482 			}});
483 		}
484 		else {
485 			node.traverseChildren({fn:function(args) {
486 				if (Util.isNotNull(this.parent) && (this.parent === this.tree.root || (this.parent._shown && !this.parent._collapsed))) {
487 					this._shown = true;
488 				}
489 				else {
490 					args['propagate'] = false;
491 					this.traverse({fn:function(args) {
492 						this._shown = false;
493 					}});
494 				}
495 			}});
496 		}
