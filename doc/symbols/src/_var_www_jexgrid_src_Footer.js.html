460 prototype._initSumCells = function() {
461 	var rows = this.grid['dataMgr'].getReal(),
462       colDefs = this.grid['colDefMgr'].get(),
463       clen = colDefs.length,
464       colDef,
465       renderer,
466       lower,
467       key,
468       name,
469       sum,
470       sumfn = Footer._calSum,
471       map = this._sumMap,
472       cell,
473       node,
474       i = 0;
475 	for (; i < clen; i++) {
476 		colDef = colDefs[i];
477 		renderer = colDef['sumRenderer'];
478 		if (Util.isNotNull(renderer)) {
479 			key = colDef['key'];
480          name = colDef['name'];
481 			sum = sumfn(rows, key);
482 			cell = map[key] = this.getNextCell();
483          node = cell[0];
484 			
485 			if (Util.isFunction(renderer)) {
486 				node.innerHTML = renderer(name, sum);
487          }
488 			else if (Util.isString(renderer)) {
489             lower = renderer.toLowerCase();
490 				if (lower === "krw" || renderer === "\\") {
491 					node.innerHTML = this._sumRenderer(name, Util.formatNumber(sum));				
492             }
493 				else if (lower === "usd" || renderer === "$") {
494 					node.innerHTML = this._sumRenderer(name, Util.formatNumber(sum, 2, "$ "));
495             }
496 			}
497 			else {
498 				node.innerHTML = this._sumRenderer(name, sum);
499          }
500 		}
501 	}

504 prototype._updateSums = function() {
505 	var rows = this.grid['dataMgr'].getReal(),
506       key,
507       map = this._sumMap,
508       cmgr = this.grid['colDefMgr'],
509       colDef,
510       renderer,
511       lower,

515       cell,
516       node,
517       content = this._options['classContent'];
518 	for (key in map) {
519       if (map.hasOwnProperty(key)) {
520          colDef = cmgr.getByKey(key);
521          name = colDef['name'];
522          renderer = colDef['sumRenderer'];
523          sum = sumfn(rows, key);
524          cell = map[key];
525          node = cell[0];
526 
527          if (Util.isFunction(renderer)) {
528             node.innerHTML = renderer(colDef['name'], sum);
529          }
530          else if (Util.isString(renderer)) {
531             lower = renderer.toLowerCase();
532             if (lower === "krw" || renderer === "\\") {
533                cell.find("span." + content)[0].innerHTML = Util.formatNumber(sum);				
534             }
535             else if (lower === "usd" || renderer === "$") {
536                cell.find("span." + content)[0].innerHTML = Util.formatNumber(sum, 2, "$ ");
537             }
538          }	
539          else {
540             cell.find("span." + content)[0].innerHTML = sum;
541          }

565 Footer._calSum = function(datalist, key) {
566 	var sum = 0,
567       tmp,
568       len = datalist.length,
569       i = 0;
570 	for (; i < len; i++) {
571 		if (typeof (tmp = datalist[i][key]) === "string") {
572 			tmp = tmp.toFloat();
573       }
574 		sum += isNaN(tmp) ? 0 : tmp;
575 	}
576 	return sum;
577 };
