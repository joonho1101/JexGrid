367 prototype._onRenderModules = function() {
368 	var html = [],
369 		colDefs = this.grid['colDefMgr'].getAll(),
370 		len = colDefs.length,
371 		colDef,
372 		opt = this._options,
373 		classCol = opt['classCol'],
374 		classColName = opt['classColName'],
375 		thisIns = this,

378 		i = 0,
379 		keyupfn = function(e) {

381 				thisIns._addData();

385 	for (; i < len; i++) {
386 		colDef = colDefs[i];
387 		if (colDef['inputOnCreate'] === true) {
388 			html.push("<div key='" + colDef['key'] + "' class='" + classCol + "'><div class='" + classColName + "'>" + colDef['name'] + "</div><input type='text' value='" + Util.ifNull(colDef['defaultValue'], "") + "' style='width:" + colDef['width'] + "px'/></div>");
389 		}
390 	}
391 
392 	creator[0].innerHTML = html.join("") + "<button type='button' onclick='JGM.m.DataCreator." + this.mid + "._addData()'>등록</button><button type='button' onclick='JGM.m.DataCreator." + this.mid + "._reset()'>초기화</button>";
393 
394 	for (i = 0; i < len; i++) {
395 		colDef = colDefs[i];
396 		if (colDef['inputOnCreate'] === true) {
397 			imap[colDef['key']] = creator.find("div[key='" + colDef['key'] + "'] input").keyup(keyupfn);

409 prototype._addData = function() {
410 	var key,
411 		imap = this._inputMap,
412 		cmgr = this.grid['colDefMgr'],
413 		colDef,
414 		input,
415 		newData = {},
416 		colDefs = cmgr.getAll(),
417 		len = colDefs.length,
418 		i = 0;
419 	for (key in imap) {
420 		if (imap.hasOwnProperty(key)) {
421 			colDef = cmgr.getByKey(key);
422 			input = imap[key];
423 		}
424 	}
425 	for (; i < len; i++) {
426 		colDef = colDefs[i];
427 		key = colDef['key'];
428 		if (imap.hasOwnProperty(key)) {
429 			newData[key] = imap[key][0].value;
430 		}
431 		else if (colDef['defaultValue'] !== undefined) {
432 			newData[key] = colDef['defaultValue'];
433 		}
434 	}
435 
436 	/**
437 	  새로운 데이터를 생성을 완료하고 그리드에 입력하기 전에 발생되는 이벤트입니다.

439 	  @event {Event} onAfterDataCreate

442 	  @author 조준호
443 	  @since 1.1.1
444 	  @version 1.1.1
445 	  */
446 	this.grid['event'].trigger("onAfterDataCreate", [newData], true);
447 	this.grid['dataMgr'].add(newData, {'isNew':true});
