150 	function BaseModule(args) {
151 		if (args) {
152 			// module ID
153 			if (args.mid != null) {
154 				this.mid = args.mid;
155 			}
156 
157 			// grid
158 			if (args.grid) {
159 				this.grid = args.grid;
160 			}
161 		}
162 
163 		// options
164 		var defaults = this._defaultOptions && this._defaultOptions(args.grid),
165 			opts = args && args['options'];
166 		if (opts || defaults) {
167 			if (!defaults) {
168 				defaults = {};
169 			}
170 			if (opts) {
171 				$.extend(true, defaults, opts);
172 			}
173 			this._options = defaults;
174 			this.dispatchEvent({
175 				'type':'afteroption',
176 				'options':defaults
177 			});
178 		}
179 
180 		// init
181 		if (this._init) {
182 			this.dispatchEvent({
183 				'type':'beforeinit'
184 			});
185 			this._init(args);
186 			this.dispatchEvent({
187 				'type':'afterinit'
188 			});
189 		}
190 
191 		var that = this,
192 			grid = this.grid;
193 		if (grid) {
194 			/*
195 			 * Common Grid Events
196 			 */
197 			['Dispose', 'CreateCss', 'CreateDynamicCss', 'Render', 'Keydown', 'Keyup', 'Keypress', 'Mousein', 'Mouseout', 'Mouseenter', 'Mouseleave', 'Mousemove', 'Mouseover', 'Mousedown', 'Mouseup', 'Click', 'Dblclick', 'Resize', 'ResizeWidth', 'ResizeHeight', 'Scroll', 'ScrollH', 'ScrollV'].forEach(function(name) {
198 				var before = 'before' + name,
199 					after = 'after' + name,
200 					beforefn = '_' + before,
201 					afterfn = '_' + after;
202 				if (that[beforefn]) {
203 					grid.addEventListener(before, function(event) {
204 						return that[beforefn](event);
205 					});
206 				}
207 				if (that[afterfn]) {
208 					grid.addEventListener(after, function(event) {
209 						return that[afterfn](event);
210 					});
211 				}
212 			});
213 		}
214 
215 		// bindEvents
216 		if (this._bindEvents) {
217 			this.dispatchEvent({
218 				'type':'beforebindevents'
219 			});
220 			this._bindEvents(args);
221 			this.dispatchEvent({
222 				'type':'afterbindevents'
223 			});
224 		}
225 
226 		// complete instantiation
227 		this.dispatchEvent({
228 			'type':'complete'
229 		});

477 	 * @param {string|object<string>} styles - string 일 경우 그대로 CSS 스타일에 추가되고, object 일 경우 key 값은 CSS 스타일 명, value 값은 CSS 스타일 값으로 추가됩니다.
478 	 * @param {boolean} noGridId - true 일 경우, 셀렉터에 그리드 아이디를 붙이지 않습니다. false 일 경우, 그리드 아이디가 셀렉터에 포함되어 그리드 컨테이너 내부의 DOM 엘레멘트에만 스타일이 적용됩니다.

487 	proto.toCssStyle = function(selector, styles, noGridId) {
488 		var css = [];
489 
490 		if (!noGridId) {
491 			selector = '#' + this.grid['mid'] + ' ' + selector;
492 		}
493 
494 		if (typeof styles != 'string') {
495 			var i,
496 				append = '';
497 
498 			if (styles.hasOwnProperty('_prepend')) {
499 				if (styles._prepend) {
500 					css.push(styles._prepend);
501 				}
502 				delete styles._prepend;
503 			}
504 
505 			if (styles.hasOwnProperty('_append')) {
506 				if (styles._append) {
507 					append = ';' + styles._append;
508 				}
509 				delete styles._append;
510 			}
511 
512 			for (i in styles) {
513 				css.push(i + ':' + styles[i]);
514 			}
515 
516 			css = css.join(';') + append;
517 		}
518 
519 		return selector + '{' + css + '}';
520 	}
