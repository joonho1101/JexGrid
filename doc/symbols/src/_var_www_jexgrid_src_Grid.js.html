558 	var link,
559 		arr,
560 		arrlen,
561 		j,
562 		path,
563 		pathlen,
564 		current,
565 		last,
566 		lastPath,
567 		i;
568 
569 	link_loop:
570 		for (link in links) {
571 			if (links.hasOwnProperty(link) && !(link in this)) {
572 				arr = Util.split(links[link]);
573 				arrlen = arr.length;
574 				j = 0;
575 
576 				arr_loop:
577 					for (; j < arrlen; j++) {
578 						path = arr[j].split(".");
579 						pathlen = path.length;
580 						if (pathlen < 1) {
581 							continue;
582 						}
583 
584 						current = this;
585 						last = this;
586 						lastPath = "";
587 						i = 0;
588 						for (; i < pathlen; i++) {
589 							if (!(path[i] in current)) {
590 								continue arr_loop;
591 							}
592 							else {
593 								last = current;
594 								current = current[lastPath = path[i]];
595 							}
596 						}
597 
598 						this._registerLink(link, current, last, lastPath);
599 						continue link_loop;
600 					}
601 			}
602 		}
603 };
