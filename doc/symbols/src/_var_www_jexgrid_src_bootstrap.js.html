 83 /**
 84   Grid 모듈을 instantiate 하고 생성된 오브젝트 인스턴스를 리턴합니다. JGM 에
 85   등록해야하는 모듈인 경우 고유의 mid를 할당하고 생성된 모듈을 {@link m} 에 모듈
 86   명을 key 값으로 그리고 mid 를 다시 한번 key 값으로 등록합니다.
 87 
 88   @function {?} create
 89   @param {string} name - Grid 모듈 이름
 90   @param {Object} args - 해당 Grid 모듈에 넘겨줄 옵션 오브젝트
 91   @returns {?} Grid 모듈 인스턴스
 92 
 93   @author 조준호
 94   @since 1.0.0
 95   @version 1.0.0
 96   */
 97 JGM.create = function(name, args) {
 98 	if (args == null) {
 99 		args = {};
100 	}
101 	if (!this.hasOwnProperty(name)) {
102 		throw new Error('cannot find a grid module: name=' + name);
103 	}
104 	if (this._map.hasOwnProperty(name)) {
105 		if (this._map[name].cacheModule) {
106 			var mid = args.mid = "JGM" + this.m.length++;
107 			var module = new this[name](args);
108 			if (!this.m.hasOwnProperty(name)) {
109 				this.m[name] = {};
110 			}
111 			this.m[name][mid] = module;
112 			if (name === "Grid" && module.name) {
113 				this.gridMap[module.name] = module;
114 			}
115 			return module;
116 		}
117 		else {
118 			return new this[name](args);
119 		}
120 	}
121 	else {
122 		return new this[name](args);
123 	}
124 };

127 JGM._destroy = function(obj, args) {
128 	var cur,
129 		arr,
130 		i,
131 		j,
132 		len;
133 	for (i in args) {
134 		if (args.hasOwnProperty(i)) {
135 			switch (i) {
136 				case "map":
137 					cur = args[i];
138 					if (Util.isString(cur)) {
139 						arr = Util.split(cur);
140 						len = arr.length;
141 						for (j = 0; j < len; j++) {
142 							JGM._deleteMap(obj, arr[j]);
143 						}
144 					}
145 					else if (cur instanceof Array) {
146 						len = cur.length;
147 						for (j = 0; j < len; j++) {
148 							Util.emptyObject(cur[j]);
149 						}
150 					}
151 					else {
152 						Util.emptyObject(cur);
153 					}
154 					break;
155 				case "array":
156 					cur = args[i];
157 					if (Util.isString(cur)) {
158 						arr = Util.split(cur);
159 						len = arr.length;
160 						for (j = 0; j < len; j++) {
161 							JGM._deleteArray(obj, arr[j]);
162 						}
163 					}
164 					else {
165 						cur.length = 0;
166 					}
167 					break;
168 				case "$":
169 					cur = args[i];
170 					if (Util.isString(cur)) {
171 						arr = Util.split(cur);
172 						len = arr.length;
173 						for (j = 0; j < len; j++) {
174 							JGM._delete$(obj, arr[j]);
175 						}
176 					}
177 					else if (cur instanceof Array) {
178 						len = cur.length;
179 						for (j = 0; j < len; j++) {
180 							Util$.unbindRemove(cur[j]);
181 						}
182 					}
183 					else {
184 						Util$.unbindRemove(cur);
185 					}
186 					break;
187 				case "style":
188 					cur = args[i];
189 					if (Util.isString(cur)) {
190 						arr = Util.split(cur);
191 						len = arr.length;
192 						for (j = 0; j < len; j++) {
193 							JGM._deleteStyle(obj, arr[j]);
194 						}
195 					}
196 					else if (cur instanceof Array) {
197 						len = cur.length;
198 						for (j = 0; j < len; j++) {
199 							Util.removeStyle(cur[j]);
200 						}
201 					}
202 					else {
203 						Util.removeStyle(cur);
204 					}
205 					break;
206 				case "property":
207 					cur = args[i];
208 					if (Util.isString(cur)) {
209 						arr = Util.split(cur);
210 						len = arr.length;
211 						for (j = 0; j < len; j++) {
212 							delete obj[arr[j]];
213 						}
214 					}
215 					else if (cur instanceof Array) {
216 						len = cur.length;
217 						for (j = 0; j < len; j++) {
218 							delete obj[cur[j]];
219 						}
220 					}
221 					break;
222 				case "module":
223 					cur = args[i];
224 					if (Util.isString(cur)) {
225 						arr = Util.split(cur);
226 						len = arr.length;
227 						for (j = 0; j < len; j++) {
228 							JGM._deleteModule(obj, arr[j]);
229 						}
230 					}
231 					else if (cur instanceof Array) {
232 						len = cur.length;
233 						for (j = 0; j < len; j++) {
234 							cur[j].destroy();
235 						}
236 					}
237 					else {
238 						cur.destroy();
239 					}
240 					break;
241 				case "name":
242 					if (obj.hasOwnProperty("mid")) {
243 						JGM._remove(args[i], obj.mid);
244 						delete obj.mid;
245 					}
246 					break;
247 				case "path":
248 					if (obj.hasOwnProperty("grid") && obj.grid.hasOwnProperty(args[i])) {
249 						delete obj.grid[args[i]];
250 						delete obj.grid;
251 					}
252 					break;
253 			}
254 		}
255 	}
256 	Util.emptyObject(obj);
257 };
