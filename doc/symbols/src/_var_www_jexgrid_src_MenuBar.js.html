 50 		'borderThickness': 1,

 70 proto._init = function(args) {

 74 		.prependTo(this._ctnr);

 76 };

 78 proto._bindEvents = function(args) {

 80 		'onCreateCss': this._onCreateCss,

 82 	}, this);
 83 }

 85 proto._destroy = function() {	
 86 	JGM._destroy(this, {
 87 		name: "MenuBar",
 88 		path: "menubar",
 89 		"$": "_menubar",
 90 		property: "_ctnr",
 91 		map: "_options"
 92 	});
 93 };

 95 proto._onCreateCss = function() {
 96 	var gridId = "#" + this.grid['mid'] + " .",
 97       opt = this._options,
 98       border = opt['borderThickness'] + "px " + opt['border'],
 99       rules = [];
100 
101 	rules.push(gridId + opt['classMenuBar'] + "{" + JGM._CONST._cssUnselectable + "overflow:hidden;width:100%;background:" + opt['background'] + ";border-bottom:" + border + ";height:" + opt['height'] + "px}");
102 	rules.push(gridId + opt['classIcon'] + "{float:left;height:" + opt['iconHeight'] + "px;width:" + opt['iconWidth'] + "px;border:" + opt['iconBorderThickness'] + "px " + opt['iconBorder'] + ";margin:" + opt['iconMargin'] + "px 0 0 " + opt['iconMargin'] + "px;background:" + opt['iconBackground'] + ";border-radius:" + opt['iconBorderRadius'] + "px;-moz-border-radius:" + opt['iconBorderRadius'] + "px}");
103 	rules.push(gridId + opt['classIcon'] + ":hover," + gridId + opt['classIcon'] + ":focus{background:" + opt['iconBackgroundHover'] + ";border:" + opt['iconBorderThickness'] + "px " + opt['iconBorderHover'] + "}");
104 	rules.push(gridId + opt['classIcon'] + ":active," + gridId + opt['classIcon'] + ".active{cursor:default;background:" + opt['iconBackgroundActive'] + ";border:" + opt['iconBorderThickness'] + "px " + opt['iconBorderActive'] + "}");
105 	rules.push(gridId + opt['classIcon'] + ".active:focus{border:" + opt['iconBorderThickness'] + "px " + opt['iconBorderFocus'] + "}");
106 	
107 	return rules.join("");
108 };
109 
110 proto.addIcon = function(css, title, width, height, fn) {
111 	return $("<div class='" + this._options['classIcon'] + "' tabIndex='0' title='" + title + "'><div class='" + css + "' style='margin-top:" + ((this._options['iconHeight'] - height) / 2) + "px;margin-left:" + ((this._options['iconWidth'] - width) / 2) + "px'></div></div>").appendTo(this._menubar)
112 	.click(function(e) {
113 		fn();
114 		$(this).toggleClass("active");
115 		e.preventDefault();
116 	})
117 	.keydown(function(e) {
118 		if (e.which === Util.keyMapKeydown.enter || e.which === Util.keyMapKeydown.space) {
119 			fn();
120 			$(this).toggleClass("active");
121 			e.preventDefault();
122 		}
123 	});
124 };
