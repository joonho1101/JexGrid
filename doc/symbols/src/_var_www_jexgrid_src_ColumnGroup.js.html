299 prototype._addData = function(data, key, idKey, notReal, nodeKey, parentKey, padColKeys, newParents) {
300 	var keyVal = data[key],
301 		parent,
302 		map = this._parentMap;
303 	if (!map.hasOwnProperty(keyVal)) {
304 		parent = this._makeParent(data, key, idKey, keyVal, notReal, nodeKey, padColKeys);
305 		newParents.push(parent);
306 		map[keyVal] = parent;
307 	}
308 	else {
309 		parent = map[keyVal];
310 	}
311 	data[nodeKey] = data[idKey];
312 	data[parentKey] = this.mid + keyVal;

364 prototype._onUpdateDatarow = function(datarow, change, before) {
365 	if (change.hasOwnProperty(this._options['key'])) {
366 		var key = this._options['key'],	// group key
367 			newkey = change[key],	// new group key value
368 			oldkey = before[key],	// old group key value
369 			mid = this.mid,	// colgroup's module id
370 			collapser = this.collapser,	// collapser used by colgroup
371 			tree = collapser._tree,	// tree used by collapser
372 			tpkey = tree._options['parentKey'],	// tree's parent key
373 			parKey = {},	// change that contains new parent key to be passed to collapser
374 			oldParKey = {},	// before that contains old parent key to be passed to collapser
375 			pmap = this._parentMap,	// group parent map
376 			oldp;	// old tree parent of currently modified record
377 
378 		// add new parent group if not exists already
379 		if (!pmap.hasOwnProperty(newkey)) {
380 			this._onAddDatarow(datarow);
381 		}
382 
383 		// pass parent key info to collapser
384 		// this will move current data's tree node from old to new parent tree node
385 		parKey[tpkey] = mid + newkey;
386 		oldParKey[tpkey] = mid + oldkey;
387 		collapser._onUpdateDatarow(datarow, parKey, oldParKey);
388 
389 		// get old parent tree node and delete the group parent if it has no children members anymore
390 		oldp = tree.getNode(pmap[oldkey]);
391 		if (oldp.children.length === 0) {
392 			this.grid['dataMgr'].all.remove(oldp.data);
393 			delete pmap[oldkey];
394 			collapser._onRemoveDatarow(oldp.data);
395 		}
396 	}

399 prototype._onUpdateDatalist = function(datalist, changes, befores) {
400 	var key = this._options['key'],	// group key
401 		mid = this.mid,	// colgroup's module id
402 		collapser = this.collapser,	// collapser used by colgroup
403 		tree = collapser._tree,	// tree used by collapser
404 		tpkey = tree._options['parentKey'],	// tree's parent key
405 		change,
406 		parKey = {},	// change that contains new parent key to be passed to collapser
407 		oldParKey = {},	// before that contains old parent key to be passed to collapser
408 		parKeys = [],	// array that contains objects with new parent keys
409 		oldParKeys = [],	// array that contains objects with old parent keys
410 		list = [],	// array that contains only datarows whose group key value has been changed 
411 		i = 0,
412 		len = datalist.length;
413 
414 	// filter list so new list only contains datarows with modified group keys
415 	for (; i < len; i++) {
416 		change = changes[i];
417 		if (change.hasOwnProperty(key)) {
418 			parKey = {};
419 			parKey[tpkey] = mid + change[key];
420 			parKeys.push(parKey);
421 			oldParKey = {};
422 			oldParKey[tpkey] = mid + befores[i][key];
423 			oldParKeys.push(oldParKey);
424 			list.push(datalist[i]);
425 		}
426 	}
427 
428 	if (list.length !== 0) {
429 		var oldkey,	// old group key value
430 			pmap = this._parentMap,	// group parent map
431 			oldp,	// old tree parent of currently modified record
432 			removeList = [];
433 
434 		// add new parent group if not exists already
435 		this._processData(list);
436 
437 		// pass parent key info to collapser
438 		// this will move current data's tree node from old to new parent tree node
439 		collapser._onUpdateDatalist(list, parKeys, oldParKeys);
440 
441 		// get old parent tree node and delete the group parent if it has no children members anymore
442 		i = 0;
443 		len = oldParKeys.length;
444 		for (; i < len; i++) {
445 			oldkey = oldParKeys[i][tpkey];
446 			if (pmap.hasOwnProperty(oldkey)) {
447 				oldp = tree.getNode(pmap[oldkey]);
448 				if (oldp.children.length === 0) {
449 					delete pmap[oldkey];
450 					removeList.push(oldp.data);
451 				}
452 			}
453 		}
454 
455 		// remove group parents with no children members
456 		if (removeList.length !== 0) {
457 			collapser._onRemoveDatalist(removeList);
458 			this.grid['dataMgr'].all.removeList(removeList);
459 		}
460 	}
461 };

463 prototype._onRemoveDatarow = function(datarow) {

465 		delete this._parentMap[datarow[this._options['key']]];

467 	else {
468 		var parentNode = this.collapser._tree.getNode(datarow).parent;
469 		if (parentNode.children.length === 1) {
470 			this.grid['dataMgr'].remove(parentNode.data);
471 		}
472 	}

483 prototype._onCollapserTreeChange = function() {
484 	var sums = {},
485 		sumKeys = this._options['sumColKeys'],
486 		len = sumKeys.length,
487 		i = 0,
488 		notReal = this.grid['dataMgr']._consts._notReal,
489 		mid = this.mid,
490 		data,
491 		curKey,
492 		tmp;
493 
494 		
495 	for (; i < len; i++) {
496 		sums[sumKeys[i]] = 0;
497 	}
498 
499 	this.collapser._tree.root.traverseChildren({post:true, fn:function(args, index) {
500 		data = this.data;
501 		i = 0;
502 		if (data[notReal] === mid) {
503 			for (; i < len; i++) {
504 				curKey = sumKeys[i];
505 				data[curKey] = sums[curKey];
506 				sums[curKey] = 0;
507 			}
508 		}
509 		else if (!data.hasOwnProperty(notReal)) {
510 			for (; i < len; i++) {
511 				if (typeof (tmp = data[sumKeys[i]]) === "string") {
512 					tmp = tmp.toFloat();				
513 				}
514 				sums[sumKeys[i]] += isNaN(tmp) ? 0 : tmp;
515 			}
516 		}
517 	}});
518 };
